    def combine(self, n: int, k: int) -> List[List[int]]:
        res = []
        path = []
        def backtrace(n,k,start):
            if len(path) == k:
                res.append(path[:])
                return
            for i in range(start,n+1):
                path.append(i)
                backtrace(n,k,i+1)
                path.pop()
        backtrace(n,k,1)
        return res
