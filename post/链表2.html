        ListNode* dummy = new ListNode();
        dummy->next = head;
        ListNode* cur = dummy;
        while(cur->next){
            ListNode* rub = cur->next;
            if(cur->next->val == val){
                cur->next = cur->next->next;
                delete rub;
            }
            else{
                cur = cur->next;
            }
        }
        return dummy->next;
    }
};
